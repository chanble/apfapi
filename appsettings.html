apf.appsettings

An element specifying the settings of the APF application.

XML Definition

<a:appsettings
     allow-blur="true | false"
     allow-select="true | false"
     auto-disable="true | false"
     auto-disable-actions="true | false"
     auto-hide-loading="true | false"
     baseurl="String"
     debug="true | false"
     default-page="String"
     disable-backspace="true | false"
     disable-f5="true | false"
     disable-right-click="true | false"
     disable-space="true | false"
     drag-outline="String | "true | false""
     empty-message="String"
     id="String"
     iepngfix="String"
     iepngfix-elements="String"
     iphone-fixed-viewport="true | false"
     iphone-fullscreen="true | false"
     iphone-icon="String"
     iphone-icon-is-glossy="true | false"
     iphone-statusbar="String"
     loading-message="String"
     login="String"
     logout="String"
     model="String"
     name="String"
     offline="String"
     offline-message="String"
     outline="String | "true | false""
     realtime="String"
     resize-outline="String | "true | false""
     skinset="String"
     storage="String"
     undokeys="true | false">
